---------

	>>> sys.path.append("..") #add parent
	>>> myList = __import__("1-my_list").MyList
	>>>

	>>> myList.append(4)
	>>> myList.append(2)
	>>> myList.append(3)
	>>> myList.append(5)
	>>> print(myList)
	[1, 4, 2, 3, 5]
